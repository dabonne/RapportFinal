	{	\def\sf@one{#1}%
		\def\sf@two{#2}%
		\setbox\sf@box\hbox
			\bgroup}%
	{	\egroup

		\def\sf@two{\@empty}
	\fi
	\ifx\@empty\sf@one\@empty\relax

	\else
		\subfloat[\sf@one][\sf@two]{\box\sf@box}%
	\fi}
